    # çŸ©é˜µä¸­çš„æœ€å°æ•°å€¼ï¼Œæœ€å¤§æ•°å€¼ï¼Œæœ€å°å€¼æ‰€åœ¨åæ ‡ï¼Œæœ€å¤§å€¼æ‰€åœ¨åæ ‡
    min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(result)

    # å¦‚æœåŸå§‹å°ºå¯¸åŒ¹é…åº¦å·²ç»å¾ˆé«˜ï¼Œç›´æ¥è¿”å›
    if max_val >= threshold:
        h, w = target_gray.shape[:2]
        center_x = max_loc[0] + w // 2
        center_y = max_loc[1] + h // 2
        return (center_x, center_y, max_loc, target_gray.shape[:2])

    # å¦‚æœåŒ¹é…åº¦æ¥è¿‘é˜ˆå€¼ï¼Œå°è¯•å¤šå°ºåº¦åŒ¹é…
    # å¯èƒ½å› ä¸ºç¼©æ”¾äº†ç•Œé¢ï¼Œæˆ–ä¸åŒåˆ†è¾¨ç‡çš„å±å¹•ä¸Šè¿è¡Œï¼Œå¯¼è‡´åŒ¹é…åº¦ä¸é«˜
    if max_val >= 0.5:
        scales = [0.95, 1.05, 0.9, 1.1]
        for scale in scales:
            h, w = target_gray.shape[:2]
            scaled_h = int(h * scale)
            scaled_w = int(w * scale)
            # æ£€æŸ¥å°ºå¯¸æ˜¯å¦æœ‰æ•ˆ
            if (
                scaled_h <= 0
                or scaled_w <= 0
                or scaled_h > screen_gray.shape[0]
                or scaled_w > screen_gray.shape[1]
            ):
                continue
            # ç¼©æ”¾æ¨¡æ¿
            scaled_target = cv2.resize(
                target_gray, (scaled_w, scaled_h), interpolation=cv2.INTER_AREA
            )
            # æ‰§è¡Œæ¨¡æ¿åŒ¹é…
            result = cv2.matchTemplate(screen_gray, scaled_target, method)
            min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(result)
            if max_val >= threshold:
                # è®¡ç®—ä¸­å¿ƒç‚¹åæ ‡,è€ƒè™‘ç¼©æ”¾åçš„å°ºå¯¸
                center_x = max_loc[0] + scaled_w // 2
                center_y = max_loc[1] + scaled_h // 2
                return (center_x, center_y, max_loc, target_gray.shape[:2])

    return None
