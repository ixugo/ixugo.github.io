	var mut sync.Mutex
	var wg sync.WaitGroup

	counter := 0
	for i := 0; i < 5000; i++ {
		wg.Add(1)
		go func() {
			mut.Lock()
			defer wg.Done()
			defer mut.Unlock()

	}

	wg.Wait()

	t.Logf("counter = %d", counter)
